library(ggplot2)
  n <- 300
  t1 <- 1:n
  t0 <- seq(3,2*n+1,2) %% n
  t2 <- t0 + (t0 == 0)*n
  
  df <- data.frame(x = cos((t1-1)*2*pi/n), 
                  y = sin((t1-1)*2*pi/n),
                  x2 = cos((t2-1)*2*pi/n),
                  y2 = sin((t2-1)*2*pi/n))
  
  ggplot(df,aes(x,y,xend = x2,yend = y2)) +
    geom_segment(alpha = .1) +
    theme_void()
