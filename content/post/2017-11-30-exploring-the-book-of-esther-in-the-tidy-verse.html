passage <- plain.text %>% 
  mutate(
        Book = str_extract(Text, "^([*?\\w]+)"), 
        Chapter = as.numeric(str_extract(Text,"\\d+(?=:\\n?)")), 
        Verse = as.numeric(str_extract(Text,"(?<=\\:)\\d+")),  
        Line = str_replace(Text, "^([*?\\w]+\\s\\w+\\:\\w+)",""))
